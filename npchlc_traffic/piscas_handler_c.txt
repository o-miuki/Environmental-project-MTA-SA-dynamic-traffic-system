-- Este evento é acionado pelo servidor (ai.lua)
addEventHandler("npchlc_traffic:onSetTurnLightState", root, 
    function(vehicle, state)
        -- Verifica se o recurso 'car_turn_lights' e a função exportada existem no cliente
        if getResourceFromName("car_turn_lights") and exports.car_turn_lights and exports.car_turn_lights.setVehicleTurnLightState then
            -- Chama a função do lado do cliente para ligar/desligar a seta
            exports.car_turn_lights:setVehicleTurnLightState(vehicle, state)
        end
    end
)
